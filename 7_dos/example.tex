%
\begin{align}
  \langle u, v \rangle & = \langle f, v\rangle\\
                       & = G(v)
\end{align}
%
versus the following
%
\begin{align}
\langle u, v \rangle & =
\langle f, v\rangle\\
& = G(v).
\end{align}
